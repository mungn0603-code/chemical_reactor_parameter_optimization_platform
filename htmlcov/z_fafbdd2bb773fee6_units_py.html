1"""core/units.py — 단위계: SI 고정 + 경계 변환. 

2 

3내부 계산은 전부 SI 단위로만 수행하고, 사용자 단위는 '입력 경계'에서 

4딱 한 번 SI로 변환한다. 각 단위는 (차원, 배율 factor, 오프셋 offset) 로 정의된다. 

5SI 값 = value * factor + offset. 

6"""

55def is_known_unit(unit: str) -> bool: 

56 """등록된 단위인지 여부를 돌려준다.""" 

57 return unit in _UNITS

59 

60def dimension_of(unit: str) -> str: 

61 """단위의 물리 차원 이름을 돌려준다. 미등록이면 UnitError.""" 

62 if unit not in _UNITS: 

63 raise UnitError(f"미등록 단위: '{unit}'. core/units.py 의 _UNITS 에 추가하세요.") 

64 return _UNITS[unit].dimension 

65 

66 

67def si_unit_of(unit: str) -> str: 

68 """해당 차원의 SI 기준 단위를 돌려준다.""" 

69 return _UNITS[unit].si_unit if unit in _UNITS else unit

71 

72def to_si(value: float, unit: str) -> float: 

73 """사용자 단위 값을 SI 값으로 변환한다.""" 

74 if unit not in _UNITS: 

75 raise UnitError(f"미등록 단위: '{unit}'") 

76 u = _UNITS[unit] 

77 return value * u.factor + u.offset 

78 

79 

80def from_si(value_si: float, unit: str) -> float: 

81 """SI 값을 지정한 사용자 단위로 되돌린다(출력 경계용).""" 

82 if unit not in _UNITS: 

83 raise UnitError(f"미등록 단위: '{unit}'") 

84 u = _UNITS[unit] 

85 return (value_si - u.offset) / u.factor
