1"""parameters/schema.py — Parameter 객체 정의. 

2 

3역할 

4---- 

5숫자 하나(값)가 아니라, 그 값이 '무엇이고 / 어떤 단위이며 / 어떤 역할이고 / 

6어떤 범위가 유효한지'를 함께 담는 하나의 묶음(Parameter)을 정의한다. 

7이것이 파라미터 주도 설계의 최소 단위이며, 자기설명과 검증의 기반이다. 

8""" 

9from __future__ import annotations 

10 

11from dataclasses import dataclass, field 

12from enum import Enum 

13from typing import Optional 

14 

15 

16class Role(str, Enum): 

17 """파라미터의 역할. 

18 

19 - INPUT : 사용자가 직접 입력하는 값 (운전조건·물성). 예: T, C_A0, A, Ea, n 

20 - CONSTANT : 바뀌면 안 되는 상수. UI에서 잠금. 예: 기체상수 R 

21 - DERIVED : 계산으로만 채워지는 결과. 사용자 입력 금지. 예: k, X, tau 

22 """ 

23 

24 INPUT = "input" 

25 CONSTANT = "constant" 

26 DERIVED = "derived" 

27 

28 

29@dataclass 

30class Parameter: 

31 """검증 가능한 하나의 물리량. 

32 

33 필드 

34 ---- 

35 key : str 

36 내부 고유 키(영문). 계산 코드가 이 키로 값을 찾는다. 

37 label : str 

38 사용자용 한글 이름. 

39 symbol : str 

40 수식 기호(예: Ea, C_A0). 

41 unit : str 

42 사용자가 입력하는 단위(core/units.py 에 등록된 것). 

43 role : Role 

44 input / constant / derived. 

45 description : str 

46 이 값이 무엇이고 왜 필요한지에 대한 한글 설명(자기설명의 핵심). 

47 value : float | None 

48 현재 값(사용자 단위 기준). derived 는 계산 후 채워진다. 

49 min, max : float | None 

50 물리적으로 허용되는 범위(사용자 단위 기준). 검증에 사용. 

51 source : str 

52 값의 출처/신뢰도 메모(문헌·가정 등). 

53 """ 

54 

55 key: str 

56 label: str 

57 symbol: str 

58 unit: str 

59 role: Role 

60 description: str 

61 value: Optional[float] = None 

62 min: Optional[float] = None 

63 max: Optional[float] = None 

64 source: str = "" 

65 # 내부 계산용 SI 값(검증 파이프라인이 채운다). 직접 설정하지 말 것. 

66 _si_value: Optional[float] = field(default=None, repr=False) 

67 

68 def __post_init__(self) -> None: 

69 # 설명 누락은 설계 위반 → 생성 시점에 바로 막는다(자기설명 강제). 

70 if not self.description or not self.description.strip(): 

71 raise ValueError( 

72 f"파라미터 '{self.key}' 에 description(한글 설명)이 없습니다. " 

73 "모든 파라미터는 설명을 가져야 합니다." 

74 ) 

75 if not isinstance(self.role, Role): 

76 self.role = Role(self.role) 

77 

78 @property 

79 def si_value(self) -> float: 

80 """검증을 통과해 SI로 변환된 값. 검증 전 접근하면 오류.""" 

81 if self._si_value is None: 

82 raise RuntimeError( 

83 f"파라미터 '{self.key}' 는 아직 검증되지 않았습니다. " 

84 "먼저 validators.validate_registry() 를 통과시키세요." 

85 ) 

86 return self._si_value
