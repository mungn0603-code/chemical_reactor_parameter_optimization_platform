16class RegistryError(KeyError): 

17 """레지스트리 조회/등록 오류."""

19 

20class ParameterRegistry: 

21 """파라미터들의 단일 저장소.""" 

22 

23 def __init__(self) -> None: 

24 self._params: dict[str, Parameter] = {} 

25 

26 def add(self, param: Parameter) -> None: 

27 """새 파라미터를 등록한다. 키 중복 시 오류.""" 

28 if param.key in self._params: 

29 raise RegistryError(f"이미 등록된 파라미터 키: '{param.key}'") 

30 self._params[param.key] = param 

31 

32 def get(self, key: str) -> Parameter: 

33 """키로 파라미터를 조회한다. 없으면 오류.""" 

34 if key not in self._params: 

35 raise RegistryError(f"등록되지 않은 파라미터: '{key}'") 

36 return self._params[key] 

37 

38 def has(self, key: str) -> bool: 

39 return key in self._params 

40 

41 def all(self) -> list[Parameter]: 

42 """등록된 모든 파라미터 목록.""" 

43 return list(self._params.values()) 

44 

45 def by_role(self, role: Role) -> list[Parameter]: 

46 """역할별 파라미터 목록(예: INPUT 만 UI에 노출할 때).""" 

47 return [p for p in self._params.values() if p.role == role] 

48 

49 def set_value(self, key: str, value: float) -> None: 

50 """사용자 입력 값을 설정한다. DERIVED 는 직접 입력 금지.""" 

51 p = self.get(key)

53 raise RegistryError(

56 p.value = value

64 p._si_value = si_value
