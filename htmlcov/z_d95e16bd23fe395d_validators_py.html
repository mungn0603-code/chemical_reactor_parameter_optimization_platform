1"""parameters/validators.py — 계산 전 6단계 검증 파이프라인. 

2 

31. 타입 검증 : 값이 숫자인가 / INPUT 인데 비어있지 않은가 

42. 단위 검증 : 단위가 등록되어 있는가(미등록 단위 거부) 

53. 범위 검증 : min <= value <= max 인가(물리적 한계) 

64. SI 환산 : 경계에서 SI로 변환해 파라미터에 저장 

75. 물리 일관성 : 파라미터 간 상호 제약(예: 온도>0K, 농도>0) 

86. (사후) 자기검증 : 계산 후 결과 타당성 검사는 reactors 쪽 check_result() 가 담당 

9""" 

10from __future__ import annotations 

11 

12from numbers import Real 

13 

14from ..core import units 

15from .registry import ParameterRegistry 

16from .schema import Parameter, Role 

17 

18 

19class ValidationError(ValueError): 

20 """검증 실패. message 에 어떤 파라미터가 왜 틀렸는지 담는다.""" 

21 

22 

23def _validate_one(p: Parameter) -> None: 

24 """파라미터 하나에 대해 1~5단계를 수행한다.""" 

25 if p.role == Role.DERIVED: 

26 return 

27 

28 if p.value is None: 

29 raise ValidationError(f"[{p.label}({p.symbol})] 값이 비어 있습니다.") 

30 if not isinstance(p.value, Real) or isinstance(p.value, bool): 

31 raise ValidationError( 

32 f"[{p.label}({p.symbol})] 숫자가 아닙니다: {p.value!r}" 

33 ) 

34 

35 if not units.is_known_unit(p.unit): 

36 raise ValidationError( 

37 f"[{p.label}({p.symbol})] 미등록 단위 '{p.unit}'. " 

38 "core/units.py 에 추가하거나 올바른 단위를 쓰세요." 

39 ) 

40 

41 v = float(p.value) 

42 if p.min is not None and v < p.min: 

43 raise ValidationError( 

44 f"[{p.label}({p.symbol})] 값 {v}{p.unit} 이 최소 {p.min}{p.unit} 보다 작습니다." 

45 ) 

46 if p.max is not None and v > p.max: 

47 raise ValidationError( 

48 f"[{p.label}({p.symbol})] 값 {v}{p.unit} 이 최대 {p.max}{p.unit} 보다 큽니다." 

49 ) 

50 

51 p._si_value = units.to_si(v, p.unit) 

52 

53 

54def _validate_physics(reg: ParameterRegistry) -> None: 

55 """5단계: 파라미터 간 물리 일관성 검사(등록된 것만 검사).""" 

56 checks = { 

57 "temperature": ("절대온도", lambda si: si > 0, "0K보다 커야 합니다"), 

58 "C_A0": ("초기 농도", lambda si: si > 0, "0보다 커야 합니다"), 

59 "v0": ("부피 유량", lambda si: si > 0, "0보다 커야 합니다"), 

60 "volume": ("반응기 부피", lambda si: si > 0, "0보다 커야 합니다"), 

61 "tau": ("체류시간", lambda si: si > 0, "0보다 커야 합니다"), 

62 } 

63 for key, (name, ok, msg) in checks.items(): 

64 if reg.has(key): 

65 p = reg.get(key) 

66 if p.role != Role.DERIVED and p._si_value is not None and not ok(p._si_value): 

67 raise ValidationError(f"[{name}] {msg}. (현재 SI값 {p._si_value})") 

68 

69 

70def validate_registry(reg: ParameterRegistry) -> None: 

71 """레지스트리 전체를 검증한다(계산 직전 반드시 호출).""" 

72 for p in reg.all(): 

73 _validate_one(p) 

74 _validate_physics(reg)
