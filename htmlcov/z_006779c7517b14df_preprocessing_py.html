25

55def suggest_preprocessing( 

56 df: pd.DataFrame, profile: Optional[DataProfile] = None 

57) -> list[PreprocessingSuggestion]: 

58 """데이터에서 전처리가 필요한 지점을 탐지하고 방법을 제안한다.""" 

59 from .profile import profile_dataframe 

60 

61 profile = profile or profile_dataframe(df) 

62 out: list[PreprocessingSuggestion] = [] 

63 

64 # 1) 중복행. 

65 if profile.n_duplicates > 0: 

66 out.append(PreprocessingSuggestion( 

67 "duplicate", None, 

68 f"완전 중복행이 {profile.n_duplicates}개 있습니다.", 

69 ["drop_duplicates", "keep"], "drop_duplicates", 

70 severity=0.7)) 

71 

72 # 2) 열별 이슈. 

73 for c in profile.columns: 

74 if c.n_missing > 0: 

75 if c.kind == "numeric": 

76 opts = ["impute_median", "impute_mean", "interpolate", 

77 "drop_missing_rows", "keep"] 

78 rec = "impute_median" 

79 else: 

80 opts = ["impute_mode", "drop_missing_rows", "keep"] 

81 rec = "impute_mode" 

82 out.append(PreprocessingSuggestion( 

83 "missing", c.name, 

84 f"{c.name} 에 결측치 {c.n_missing}개({c.missing_ratio * 100:.1f}%) 발견.", 

85 opts, rec, severity=min(1.0, 0.5 + c.missing_ratio))) 

86 

87 if c.kind == "numeric" and c.n_outliers > 0: 

88 out.append(PreprocessingSuggestion( 

89 "outlier", c.name, 

90 f"{c.name} 에 이상치 {c.n_outliers}개 발견(IQR 기준).", 

91 ["winsorize", "remove_outliers", "keep"], "winsorize", 

92 severity=0.6)) 

93 

94 if c.kind == "numeric" and c.n_negative > 0: 

95 out.append(PreprocessingSuggestion( 

96 "negative", c.name, 

97 f"{c.name} 에 음수 값 {c.n_negative}개(물리적으로 타당한지 확인).", 

98 ["keep", "clip_negative", "abs_negative"], "keep", 

99 severity=0.4)) 

100 

101 if (c.kind == "numeric" and c.skew is not None and c.skew >= 1.0 

102 and c.min is not None and c.min > 0): 

103 out.append(PreprocessingSuggestion( 

104 "skew", c.name, 

105 f"{c.name} 분포가 치우쳐 있습니다(왜도={c.skew:.2f}).", 

106 ["log_transform", "keep"], "log_transform", severity=0.4)) 

107 

108 # 3) 스케일 차이(정규화/표준화 제안). 

109 if len(profile.numeric_cols) >= 2: 

110 stds = [c.std for c in profile.columns if c.kind == "numeric" and c.std] 

111 if stds and max(stds) > 0 and (max(stds) / (min(stds) or 1e-9)) > 100: 

112 out.append(PreprocessingSuggestion( 

113 "scale", None, 

114 "변수 간 스케일 차이가 큽니다. 거리 기반 분석(PCA·클러스터링) 전 스케일 조정 권장.", 

115 ["standardize", "normalize", "keep"], "standardize", severity=0.5)) 

116 

117 out.sort(key=lambda s: s.severity, reverse=True) 

118 return out

121def apply_action( 

122 df: pd.DataFrame, action: str, column: Optional[str] = None, **kwargs: Any 

123) -> pd.DataFrame: 

124 """전처리 action 을 새 데이터프레임에 적용해 반환한다(원본 불변).""" 

125 work = df.copy() 

126 

127 if action == "keep": 

128 return work 

129 if action == "drop_duplicates": 

130 return work.drop_duplicates().reset_index(drop=True) 

131 

132 if column is None: 

133 raise ValueError(f"'{action}' 에는 대상 column 이 필요합니다.") 

134 if column not in work.columns: 

135 raise KeyError(column) 

136 

137 s = work[column] 

138 

139 if action == "impute_mean": 

140 work[column] = s.fillna(pd.to_numeric(s, errors="coerce").mean()) 

141 elif action == "impute_median": 

142 work[column] = s.fillna(pd.to_numeric(s, errors="coerce").median()) 

143 elif action == "impute_mode": 

144 mode = s.mode(dropna=True) 

145 work[column] = s.fillna(mode.iloc[0] if len(mode) else s) 

146 elif action == "impute_value": 

147 work[column] = s.fillna(kwargs["value"]) 

148 elif action == "interpolate": 

149 work[column] = pd.to_numeric(s, errors="coerce").interpolate( 

150 method="linear", limit_direction="both") 

151 elif action == "drop_missing_rows": 

152 work = work[work[column].notna()].reset_index(drop=True) 

153 elif action == "remove_outliers": 

154 res = detect_outliers(work, [column], method=kwargs.get("method", "iqr")) 

155 work = work[~res.mask].reset_index(drop=True) 

156 elif action == "winsorize": 

157 work[column] = _winsorize(pd.to_numeric(s, errors="coerce"), 

158 kwargs.get("k", 1.5)) 

159 elif action == "clip_negative": 

160 work[column] = pd.to_numeric(s, errors="coerce").clip(lower=0) 

161 elif action == "abs_negative": 

162 work[column] = pd.to_numeric(s, errors="coerce").abs() 

163 elif action == "log_transform": 

164 num = pd.to_numeric(s, errors="coerce") 

165 work[column] = np.log1p(num.clip(lower=0)) 

166 elif action == "normalize": 

167 num = pd.to_numeric(s, errors="coerce") 

168 rng = num.max() - num.min() 

169 work[column] = (num - num.min()) / (rng if rng else 1.0) 

170 elif action == "standardize": 

171 num = pd.to_numeric(s, errors="coerce") 

172 std = num.std() 

173 work[column] = (num - num.mean()) / (std if std else 1.0) 

174 elif action == "encode_label": 

175 work[column] = s.astype("category").cat.codes 

176 elif action == "encode_onehot": 

177 dummies = pd.get_dummies(s, prefix=column) 

178 work = pd.concat([work.drop(columns=[column]), dummies], axis=1) 

179 else: 

180 raise ValueError(f"알 수 없는 action: {action!r}") 

181 

182 return work

185def _winsorize(s: pd.Series, k: float = 1.5) -> pd.Series: 

186 """IQR 경계 밖 값을 경계값으로 캡(Winsorizing)한다.""" 

187 q1, q3 = s.quantile(0.25), s.quantile(0.75) 

188 iqr = q3 - q1 

189 if pd.isna(iqr) or iqr == 0: 

190 return s 

191 lo, hi = q1 - k * iqr, q3 + k * iqr 

192 return s.clip(lower=lo, upper=hi)
