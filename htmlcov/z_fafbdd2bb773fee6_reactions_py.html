1"""core/reactions.py — 반응식 → 화학종 매핑 → 표준 반응 열역학 자동 계산. 

2 

3교과서 이론(표준상태 298.15 K) 

4------------------------------ 

5표준 생성엔탈피 ΔHf°, 표준 몰엔트로피 S° 로부터 Hess 의 법칙으로 반응의 

6표준 열역학량을 계산한다(Smith·Van Ness·Abbott, *Introduction to Chemical 

7Engineering Thermodynamics*; Fogler, *Elements of CRE* 부록 참조). 

8 

9 ΔH_rxn° = Σ ν_i · ΔHf°_i (생성물 ν>0, 반응물 ν<0) [Hess 의 법칙] 

10 ΔS_rxn° = Σ ν_i · S°_i 

11 ΔG_rxn° = ΔH_rxn° − T · ΔS_rxn° (ΔH°, ΔS° 는 298.15 K 값으로 근사) 

12 K_eq = exp(−ΔG_rxn° / (R·T)) 

13 

14이 값들은 기존 ThermoAnalyzer(dH, dS)·EnergyBalance(dH_rxn) 에 그대로 주입되어 

15계산 로직을 전혀 바꾸지 않고 '반응식 입력 → 자동 설정' 을 가능하게 한다. 

16 

17데이터 출처: NIST Chemistry WebBook / CRC Handbook / Smith·Van Ness·Abbott 부록 C 

18의 널리 통용되는 표준값(298.15 K). 값은 species DB 에 명시되어 추적 가능하다. 

19"""

116def parse_formula(formula: str) -> dict[str, int]: 

117 """화학식을 원소 개수 dict 로 파싱한다(괄호 미지원, 단순 조성식). 

118 

119 예: 'C2H5OH' 로 쓰지 말고 조성식 'C2H6O' 를 쓴다(Species.formula 규칙). 

120 """ 

121 counts: dict[str, int] = {} 

122 for elem, num in _ELEMENT.findall(formula): 

123 if not elem: 

124 continue 

125 counts[elem] = counts.get(elem, 0) + (int(num) if num else 1) 

126 return counts

142@dataclass 

143class Reaction: 

144 """파싱된 반응. 계수는 반응물 음수, 생성물 양수로 통일한다.""" 

145 

146 reactants: list[tuple[float, Species]] # (계수>0, 화학종) 

147 products: list[tuple[float, Species]] 

148 raw: str 

149 

150 def stoich(self) -> list[tuple[float, Species]]: 

151 """부호 있는 계수 목록(ν<0 반응물, ν>0 생성물).""" 

152 return [(-c, s) for c, s in self.reactants] + [(c, s) for c, s in self.products] 

153 

154 def equation(self) -> str: 

155 """사람이 읽는 균형 반응식 문자열.""" 

156 def side(terms): 

157 return " + ".join( 

158 (f"{c:g} {s.key}" if abs(c - 1.0) > 1e-9 else s.key) 

159 for c, s in terms) 

160 return f"{side(self.reactants)} → {side(self.products)}" 

161 

162 

163def parse_reaction(text: str) -> Reaction: 

164 """'N2 + 3 H2 -> 2 NH3' 같은 문자열을 Reaction 으로 파싱한다.""" 

165 if not _ARROW.search(text): 

166 raise ReactionError("반응식에 화살표(-> 또는 =)가 없습니다. 예: 'N2 + 3 H2 -> 2 NH3'") 

167 lhs, rhs = _ARROW.split(text, maxsplit=1) 

168 

169 def parse_side(side: str) -> list[tuple[float, Species]]: 

170 terms: list[tuple[float, Species]] = [] 

171 for part in side.split("+"): 

172 part = part.strip() 

173 if not part: 

174 continue 

175 m = _TERM.match(part) 

176 if not m: 

177 raise ReactionError(f"항을 해석할 수 없습니다: '{part}'") 

178 coeff_str, token = m.group(1), m.group(2) 

179 coeff = float(coeff_str) if coeff_str else 1.0 

180 terms.append((coeff, _lookup(token))) 

181 return terms 

182 

183 reactants = parse_side(lhs) 

184 products = parse_side(rhs) 

185 if not reactants or not products: 

186 raise ReactionError("반응물 또는 생성물이 비어 있습니다.") 

187 return Reaction(reactants=reactants, products=products, raw=text.strip())

190def element_balance(rxn: Reaction) -> tuple[bool, dict[str, float]]: 

191 """원소 균형을 검사한다. (균형 여부, 원소별 (생성물−반응물) 잔차).""" 

192 diff: dict[str, float] = {} 

193 for coeff, sp in rxn.stoich(): # coeff 부호 포함 

194 for elem, num in parse_formula(sp.formula).items(): 

195 diff[elem] = diff.get(elem, 0.0) + coeff * num 

196 balanced = all(abs(v) < 1e-9 for v in diff.values()) 

197 residual = {e: v for e, v in diff.items() if abs(v) > 1e-9} 

198 return balanced, residual

204@dataclass 

205class ReactionThermo: 

206 """반응식으로부터 자동 계산된 표준 반응 열역학량.""" 

207 

208 reaction: Reaction 

209 dH_rxn: float # 표준 반응 엔탈피 [kJ/mol] 

210 dS_rxn: float # 표준 반응 엔트로피 [J/mol/K] 

211 dG_rxn: float # 표준 반응 깁스에너지 [kJ/mol] (온도 T) 

212 K_eq: float # 평형상수 (온도 T) 

213 T: float # 계산에 쓴 온도 [K] 

214 balanced: bool 

215 residual: dict[str, float] = field(default_factory=dict) 

216 steps: list[str] = field(default_factory=list) 

217 

218 @property 

219 def enthalpy_label(self) -> str: 

220 """발열/흡열 판정(교과서 정의).""" 

221 if self.dH_rxn < 0: 

222 return "발열(exothermic)" 

223 if self.dH_rxn > 0: 

224 return "흡열(endothermic)" 

225 return "열중립" 

226 

227 @property 

228 def spontaneity_label(self) -> str: 

229 """ΔG 부호로 자발성 판정.""" 

230 if self.dG_rxn < 0: 

231 return "자발적" 

232 if self.dG_rxn > 0: 

233 return "비자발적" 

234 return "평형"

237def reaction_thermochemistry(rxn: Reaction, T: float = T_STD) -> ReactionThermo: 

238 """Hess 의 법칙으로 ΔH_rxn°, ΔS_rxn°, ΔG_rxn°, K_eq 를 계산한다. 

239 

240 ΔH°, ΔS° 는 298.15 K 표준값을 사용하고, ΔG(T)=ΔH°−T·ΔS° 로 온도만 반영한다 

241 (표준상태 근사 — 교과서 기본형). T 는 절대온도[K]. 

242 """ 

243 if T <= 0: 

244 raise ReactionError("절대온도 T 는 0 K 보다 커야 합니다.") 

245 

246 balanced, residual = element_balance(rxn) 

247 

248 steps: list[str] = [] 

249 # ΔH_rxn° = Σ ν·ΔHf° 

250 h_terms = [] 

251 dH = 0.0 

252 for coeff, sp in rxn.stoich(): 

253 dH += coeff * sp.dHf298 

254 h_terms.append(f"({coeff:+g})·({sp.dHf298:g})") 

255 steps.append("ΔH_rxn° = Σν·ΔHf° = " + " + ".join(h_terms) + f" = {dH:.4g} kJ/mol") 

256 

257 # ΔS_rxn° = Σ ν·S° 

258 s_terms = [] 

259 dS = 0.0 

260 for coeff, sp in rxn.stoich(): 

261 dS += coeff * sp.S298 

262 s_terms.append(f"({coeff:+g})·({sp.S298:g})") 

263 steps.append("ΔS_rxn° = Σν·S° = " + " + ".join(s_terms) + f" = {dS:.4g} J/mol·K") 

264 

265 # ΔG(T) = ΔH° − T·ΔS° (ΔH° kJ, ΔS° J → kJ 로 환산) 

266 dG = dH - T * (dS / 1000.0) 

267 steps.append( 

268 f"ΔG_rxn°(T={T:.2f}K) = ΔH° − T·ΔS° = {dH:.4g} − {T:.2f}·({dS:.4g}/1000) " 

269 f"= {dG:.4g} kJ/mol") 

270 

271 # K_eq = exp(−ΔG/RT) (ΔG 를 J/mol 로). 지수가 매우 크면 float 범위를 넘으므로 보호. 

272 exponent = -(dG * 1000.0) / (R_GAS * T) 

273 K_eq = _safe_exp(exponent) 

274 steps.append(f"K_eq = exp(−ΔG/RT) = exp({exponent:.4g}) = {K_eq:.4g}") 

275 

276 return ReactionThermo( 

277 reaction=rxn, dH_rxn=dH, dS_rxn=dS, dG_rxn=dG, K_eq=K_eq, T=T, 

278 balanced=balanced, residual=residual, steps=steps) 

279 

280 

281def analyze_reaction(text: str, T: float = T_STD) -> ReactionThermo: 

282 """반응식 문자열 하나로 파싱→매핑→열역학 계산을 한 번에 수행한다.""" 

283 return reaction_thermochemistry(parse_reaction(text), T)
