1"""eda/sample.py — 데모용 반응기 데이터셋 생성. 

2 

3EDA Assistant 를 바로 체험할 수 있도록, 실제 계산 엔진(CSTR)을 이용해 온도·압력을 

4스윕한 합성 데이터셋을 만든다. 일부 결측치·중복·이상치를 일부러 섞어 전처리 기능을 

5시연할 수 있게 한다. 

6""" 

7from __future__ import annotations 

8 

9import numpy as np 

10import pandas as pd 

11 

12 

13def make_sample_dataset(n: int = 120, seed: int = 7) -> pd.DataFrame: 

14 """반응기 운전 데이터를 흉내 낸 데모 데이터프레임을 만든다. 

15 

16 계산 엔진을 실제로 호출하되, 실패해도 되도록 순수 수식 근사로 폴백한다. 

17 """ 

18 rng = np.random.default_rng(seed) 

19 temps = rng.uniform(40, 120, n) # °C 

20 pressures = rng.uniform(1, 10, n) # bar 

21 residence = rng.uniform(30, 300, n) # s 

22 ea = rng.normal(50, 3, n) # kJ/mol 

23 

24 # Arrhenius 근사로 전환율 계산(엔진과 무관한 데모용 근사). 

25 R = 8.314 

26 A = 1e6 

27 k = A * np.exp(-(ea * 1000) / (R * (temps + 273.15))) 

28 x = 1 - 1 / (1 + k * residence) # 0~1 범위 근사 

29 x = np.clip(x + rng.normal(0, 0.02, n), 0, 1) 

30 yield_ = np.clip(x * rng.uniform(0.85, 0.98, n), 0, 1) 

31 profit = x * pressures * 1000 - residence * 2 + rng.normal(0, 50, n) 

32 

33 catalyst = rng.choice(["Pt", "Pd", "Ni"], n) 

34 

35 df = pd.DataFrame({ 

36 "Temperature": np.round(temps, 2), 

37 "Pressure": np.round(pressures, 2), 

38 "ResidenceTime": np.round(residence, 1), 

39 "ActivationEnergy": np.round(ea, 2), 

40 "Conversion": np.round(x, 4), 

41 "Yield": np.round(yield_, 4), 

42 "Profit": np.round(profit, 1), 

43 "Catalyst": catalyst, 

44 }) 

45 

46 # 전처리 시연을 위해 일부러 흠집을 낸다. 

47 miss_idx = rng.choice(n, size=max(1, n // 25), replace=False) 

48 df.loc[miss_idx, "Temperature"] = np.nan 

49 # 이상치 몇 개 주입. 

50 out_idx = rng.choice(n, size=3, replace=False) 

51 df.loc[out_idx, "Pressure"] = df["Pressure"].max() * 5 

52 # 중복행 몇 개. 

53 df = pd.concat([df, df.iloc[:3]], ignore_index=True) 

54 

55 return df
